## WHY
In TypeScript, you often have a value that could be one of several types — for example, `string | number | null`. Before you can safely call string methods on it (like `.toUpperCase()`), TypeScript demands you prove it's actually a string at that point in the code.

This process is called **Type Narrowing** — using runtime JavaScript checks to "narrow" the type of a variable from a broader Union type to a more specific type. TypeScript's control flow analysis is smart enough to understand these checks and automatically restrict the type within each code branch.

## THEORY
### Control Flow Analysis
TypeScript performs "Control Flow Analysis" (CFA) on your code. When you write `if (typeof value === 'string')`, TypeScript understands that inside the `if` block, `value` is guaranteed to be a `string`. This is not magic — TypeScript statically analyses the flow of your program.

## CODE

### Level 1 — Beginner (typeof Narrowing)
```typescript
function formatValue(value: string | number): string {
    // 'value' here is still 'string | number'

    if (typeof value === "string") {
        // TypeScript NARROWS: inside this block, 'value' is 'string'
        return value.toUpperCase();
    }
    
    // TypeScript NARROWS: outside the if block, 'value' MUST be 'number'
    return value.toFixed(2);
}

formatValue("hello"); // "HELLO"
formatValue(3.14159); // "3.14"
```

### Level 2 — Building Blocks (Truthiness & null Narrowing)
```typescript
function processText(text: string | null | undefined): string {
    // Falsy check — narrows away null and undefined
    if (!text) {
        return "No text provided";
    }
    
    // 'text' is now narrowed to just 'string' here!
    return text.trim().toUpperCase();
}

// Nullish coalescing operator (??) as a shorthand for null/undefined narrowing
function getDisplayName(name: string | null): string {
    return name ?? "Anonymous"; // Return 'name' if not null/undefined
}
```

### Level 3 — Intermediate (instanceof & Type Predicates)
```typescript
class NetworkError extends Error {
    statusCode: number;
    constructor(message: string, code: number) {
        super(message);
        this.statusCode = code;
    }
}

function handleError(error: unknown): void {
    if (error instanceof NetworkError) {
        // Narrowed to NetworkError — can access .statusCode!
        console.error(`HTTP ${error.statusCode}: ${error.message}`);
    } else if (error instanceof Error) {
        // Narrowed to Error — can access .message!
        console.error(error.message);
    } else {
        console.error("Unknown error:", error);
    }
}

// User-Defined Type Guards (Type Predicates)
// The return type 'value is Dog' tells TypeScript: 
// "If this function returns true, narrow the type to Dog!"
interface Dog { breed: string; bark(): void; }
interface Cat { color: string; meow(): void; }

function isDog(animal: Dog | Cat): animal is Dog {
    return (animal as Dog).breed !== undefined;
}

function makeNoise(animal: Dog | Cat): void {
    if (isDog(animal)) {
        animal.bark(); // TypeScript knows it's a Dog!
    } else {
        animal.meow(); // TypeScript knows it's a Cat!
    }
}
```

### Level 4 — Advanced (Discriminated Unions)
The most powerful and ergonomic narrowing pattern in TypeScript is the **Discriminated Union** — a Union of types that all share a common "tag" property with a unique literal value.

```typescript
// Each type has a unique 'type' literal tag
interface PendingOrder { type: "PENDING"; orderId: number; }
interface ProcessingOrder { type: "PROCESSING"; orderId: number; trackingId: string; }
interface CompletedOrder { type: "COMPLETED"; orderId: number; deliveredAt: Date; }
interface FailedOrder { type: "FAILED"; orderId: number; errorCode: string; }

type Order = PendingOrder | ProcessingOrder | CompletedOrder | FailedOrder;

function displayOrderStatus(order: Order): string {
    // TypeScript narrows based on the 'type' field!
    switch (order.type) {
        case "PENDING":
            return `Order ${order.orderId} is waiting to be processed.`;
        case "PROCESSING":
            // 'order' is now 'ProcessingOrder' — has .trackingId!
            return `Order ${order.orderId} is being shipped. Tracking: ${order.trackingId}`;
        case "COMPLETED":
            // 'order' is now 'CompletedOrder' — has .deliveredAt!
            return `Order ${order.orderId} delivered on ${order.deliveredAt.toDateString()}`;
        case "FAILED":
            // 'order' is now 'FailedOrder' — has .errorCode!
            return `Order ${order.orderId} failed! Error: ${order.errorCode}`;
    }
}
```

## REAL_WORLD
Discriminated Unions are the correct TypeScript pattern for modeling API response states. Instead of having a chaotic object where every field could be `null` at any time, you create explicit states.

```typescript
// Anti-Pattern: chaotic, hard to reason about
interface FetchResult {
    data: User | null;
    error: string | null;
    isLoading: boolean;
}

// Better: Discriminated Union — each state is clear and complete
type FetchState<T> =
    | { status: "loading" }
    | { status: "error"; message: string }
    | { status: "success"; data: T };

// The UI component can safely narrow the state:
function UserComponent({ state }: { state: FetchState<User> }) {
    if (state.status === "loading") return <Spinner />;
    if (state.status === "error") return <ErrorMessage msg={state.message} />;
    return <Profile user={state.data} />; // TypeScript knows data is User!
}
```

## INTERVIEW
**Q: What is the `never` type and how does it relate to narrowing?**
A: `never` represents a state that should be impossible. In an exhaustive switch statement with a Discriminated Union, the `default` case can have a variable typed as `never`, meaning "if we reach this case, it is logically impossible." If you add a new variant to the Union without handling it in the switch, TypeScript will throw a compile error because `never` is now being assigned a real value, proving the switch is no longer exhaustive.

**Q: What does the `as` keyword do and when should you avoid it?**
A: `as` is a "Type Assertion" — you tell TypeScript "Trust me, I know this is a `User`." TypeScript performs no runtime check; it simply removes the type warning. This is dangerous. If you're wrong, you get a silent runtime crash. The `as` keyword should be a last resort. Always prefer Type Guards (narrowing) over Type Assertions.

## FEYNMAN CHECK

### Explain Type Narrowing Like I'm 10 Years Old
> Narrowing is when TS WATCHES your runtime checks and TIGHTENS the variable's type within that branch. You start with `x: string | number`, you write `if (typeof x === 'string')`, and inside the if-block TS now treats x as `string` — autocomplete, no errors. Outside the if, x goes back to the union. The narrowers: `typeof` (primitives), `instanceof` (classes), `in` operator (property presence), discriminant property check (kind === 'ok'), `Array.isArray`, custom type guards `function isX(v): v is X`, and assertion functions `function assertX(v): asserts v is X`. The non-obvious power: narrowing flows through assignment, so `const y = x; if (typeof y === 'string')` narrows y, not x — copy semantics.

### 5 Deep Conceptual Questions

**Q1: Why are type guards safer than type casts?**
> **A:** A type cast (`x as User`) is a PROMISE to the compiler — "trust me, this is a User" — with zero runtime validation. A type guard (`function isUser(x): x is User`) is a FUNCTION that returns a boolean — and IF it returns true, TS narrows the type. The guard's body actually CHECKS the shape at runtime. Casts cause silent runtime crashes; guards convert "I think" into "I verified."

**Q2: Mental model for narrowing?**
> **A:** Think of TS as running a flow analysis: at every line, it has a current TYPE for every variable. Each runtime check NARROWS the type within the branch where the check is true. Assignments WIDEN it back. Closures may invalidate narrowing (the variable could change later). The narrowing is purely AT COMPILE TIME — at runtime your code looks identical to what you wrote.

**Q3: Most dangerous misconception? Show with code.**
> **A:** Believing narrowing persists through closures:
> ```ts
> // ❌ Narrowing is lost in the callback
> function setup(value: string | null) {
>   if (value !== null) {
>     setTimeout(() => console.log(value.length));   // TS error: 'value' possibly null
>   }
> }
>
> // ✅ Capture in a const that can't be reassigned
> function setup(value: string | null) {
>   if (value !== null) {
>     const v = value;   // v is provably string
>     setTimeout(() => console.log(v.length));   // OK
>   }
> }
> ```

**Q4: How do assertion functions differ from type guards?**
> **A:** A type guard returns boolean and narrows in the if/else. An assertion function (`function assert(c): asserts c`) THROWS if the check fails — and afterward TS knows the type is narrowed for the REST of the scope. Use guards when you want to handle both paths; assertions when failure means program is broken (e.g., invariants, precondition checks). Assertion functions enable `assert(user); user.foo` patterns without nesting.

**Q5: FAANG-grade definition?**
> **A:** "Type narrowing is TypeScript's control-flow analysis algorithm that progressively refines a variable's type within a code branch based on type guards (typeof, instanceof, in, discriminant equality, user-defined predicates, assertion functions) — purely a compile-time mechanism with zero runtime cost, governed by definite-assignment analysis and invalidated by closures, reassignments, and function boundaries that may mutate the variable."

#### Step 2 — Type Guards Library
```ts
// src/guards.ts
export function isString(v: unknown): v is string {
  return typeof v === 'string';
}
export function isNumber(v: unknown): v is number {
  return typeof v === 'number' && !Number.isNaN(v);
}
export function isObject(v: unknown): v is Record<string, unknown> {
  return typeof v === 'object' && v !== null && !Array.isArray(v);
}
export function isArrayOf<T>(v: unknown, item: (x: unknown) => x is T): v is T[] {
  return Array.isArray(v) && v.every(item);
}
export function hasProp<K extends string>(v: unknown, key: K): v is Record<K, unknown> {
  return isObject(v) && key in v;
}
export function assertString(v: unknown): asserts v is string {
  if (!isString(v)) throw new TypeError('expected string, got ' + typeof v);
}
```

#### Step 3 — Compose Guards For a Real Shape
```ts
// src/main.ts
import { isString, isNumber, isObject, isArrayOf, hasProp } from './guards.js';

interface User { id: number; name: string; emails: string[] }

function isUser(v: unknown): v is User {
  return isObject(v)
    && hasProp(v, 'id')     && isNumber(v.id)
    && hasProp(v, 'name')   && isString(v.name)
    && hasProp(v, 'emails') && isArrayOf(v.emails, isString);
}

function parseUser(json: string): User {
  const data: unknown = JSON.parse(json);
  if (!isUser(data)) throw new Error('JSON does not match User shape');
  return data;   // narrowed to User from this point
}

console.log(parseUser('{"id":1,"name":"Ana","emails":["a@b.io"]}'));
```

#### Step 4 — Error Handling: Assertion Function For Invariants
```ts
function assertUser(v: unknown): asserts v is User {
  if (!isUser(v)) throw new Error('not a User: ' + JSON.stringify(v));
}

function process(raw: unknown) {
  assertUser(raw);
  // From here, TS knows raw is User WITHOUT nesting
  console.log(raw.name.toUpperCase(), raw.emails.length);
}

try { process({ id: 1, name: 'A', emails: [] }); } catch (e) { console.error(e); }
try { process({ id: 'oops' }); }                  catch (e) { console.error('caught'); }
```

#### Step 5 — Tests
```ts
import { describe, it, expect } from 'vitest';
import { isString, isArrayOf, isObject } from './guards.js';

describe('guards', () => {
  it('isString narrows', () => {
    const x: unknown = 'hello';
    expect(isString(x)).toBe(true);
    if (isString(x)) expect(x.toUpperCase()).toBe('HELLO');
  });
  it('isArrayOf composes', () => {
    expect(isArrayOf(['a', 'b'], isString)).toBe(true);
    expect(isArrayOf(['a', 1],   isString)).toBe(false);
  });
  it('isObject rejects null and arrays', () => {
    expect(isObject(null)).toBe(false);
    expect(isObject([1])).toBe(false);
    expect(isObject({ a: 1 })).toBe(true);
  });
});
```
