## WHY
In the real world, things fail constantly. A database server goes offline, a user inputs invalid JSON, or a third-party API rate-limits your application.

If an error occurs in JavaScript and you have not written code to handle it, the error "bubbles up" to the global environment. In the browser, this results in a silent red error in the DevTools console while the user stares at a broken, frozen UI. In Node.js, an unhandled exception will cause the entire server process to violently crash, disconnecting every single user currently logged into the system. 
Mastering Error Handling allows you to intercept crashes, display friendly warning messages to users, and keep your applications running smoothly despite unpredictable failures.

## THEORY
### The Call Stack and Bubbling
When an Error is "Thrown", the JavaScript engine stops executing the current function and looks for a safety net (a `catch` block). If it doesn't find one, it travels backwards down the Call Stack to the function that called it. It keeps traveling backwards, destroying the execution state of every function it touches, until it either finds a `catch` block or hits the global environment and crashes.

### Try / Catch / Finally
The standard architecture for handling errors:
- **`try`**: A block of code that you suspect *might* fail. JS executes this carefully.
- **`catch`**: The safety net. If any code inside the `try` block throws an error, execution instantly teleports to the `catch` block.
- **`finally`**: An optional block that will execute with 100% mathematical certainty, regardless of whether the `try` succeeded or failed. (Crucial for cleanup tasks like closing database connections).

## CODE

### Level 1 — Beginner (The Basic Try/Catch)
```javascript
// 1. A Dangerous Operation
// JSON.parse will CRASH if the string is not perfectly formatted JSON!
const badData = '{"name": "Alice", age: 25}'; // Missing quotes around 'age'!

try {
  console.log("1. Attempting to parse data...");
  const user = JSON.parse(badData); // The crash happens exactly here!
  
  // This line will NEVER EXECUTE. 
  // Execution teleported away the millisecond the error occurred.
  console.log("2. Data parsed successfully!", user); 
  
} catch (error) {
  // We caught the crash! The app survives!
  console.error("3. CAUGHT AN ERROR:", error.message);
  // Output: "CAUGHT AN ERROR: Expected property name or '}' in JSON at position 19"
}

console.log("4. The application continues running perfectly!");
```

### Level 2 — Building Blocks (Throwing Custom Errors)
```javascript
// You don't have to wait for JS to throw an error. You can throw your own!

function withdrawMoney(balance, amount) {
  if (typeof amount !== "number") {
    // We throw a brand new Error Object!
    // This instantly halts the function.
    throw new TypeError("Amount must be a number!"); 
  }
  
  if (amount > balance) {
    // You can throw standard Errors, or specific types!
    throw new Error("Insufficient funds for this transaction.");
  }
  
  return balance - amount;
}

try {
  withdrawMoney(100, 500);
} catch (error) {
  // We intercept the custom error and show it to the user gracefully!
  console.log(`TRANSACTION FAILED: ${error.message}`);
}
```

### Level 3 — Intermediate (The Finally Block)
```javascript
// Imagine we are writing to a sensitive file. We open the file, write data, and close it.
// If an error happens during writing, the function crashes BEFORE the close command!
// The file is permanently locked!

let fileIsLocked = false;

function processFile() {
  try {
    fileIsLocked = true;
    console.log("File opened. Writing data...");
    
    // Simulate a random crash!
    throw new Error("Disk Full!");
    
  } catch (error) {
    console.error("Write failed:", error.message);
    
  } finally {
    // THE MAGIC: This block executes NO MATTER WHAT HAPPENS.
    // Even if you put a 'return' inside the try or catch, 'finally' overrides it 
    // and forces its code to run before the function is allowed to return!
    fileIsLocked = false;
    console.log("Cleanup complete. File lock released.");
  }
}

processFile();
console.log(`Is file still locked? ${fileIsLocked}`); // false
```

### Level 4 — Advanced (Custom Error Classes)
```javascript
// In large Enterprise apps, catching a generic 'Error' is bad practice.
// You want to know EXACTLY what failed (Network? Database? Validation?)

// We build Custom Error Classes by extending the built-in Error object!
class ValidationError extends Error {
  constructor(message, field) {
    super(message);
    this.name = "ValidationError"; // Identify the error type
    this.field = field;            // Attach custom data!
  }
}

class DatabaseError extends Error {
  constructor(message, query) {
    super(message);
    this.name = "DatabaseError";
    this.query = query;
  }
}

function saveUser(username) {
  try {
    if (username.length < 3) {
      throw new ValidationError("Username too short!", "username_input");
    }
    // Simulate DB failure
    throw new DatabaseError("Connection timed out.", "INSERT INTO users...");
    
  } catch (error) {
    // We can conditionally handle errors based on their exact Class!
    if (error instanceof ValidationError) {
      console.log(`UI ALERT: Fix the ${error.field} field. Reason: ${error.message}`);
    } else if (error instanceof DatabaseError) {
      console.log(`SYSADMIN ALERT: DB failed running ${error.query}`);
    } else {
      console.log("UNKNOWN FATAL CRASH!");
    }
  }
}

saveUser("Al"); // Triggers ValidationError handling
```

## REAL_WORLD
When fetching data from an external API using `fetch()` or `axios`, wrapping the request in a `try/catch` is mandatory. 
If the user is on a mobile phone and drives into a tunnel, their internet connection drops. The API request will violently crash with a `NetworkError`. 

```javascript
async function loadDashboard() {
  try {
    // Turn on a loading spinner in the UI
    setLoading(true);
    
    const data = await fetchUserData();
    renderDashboard(data);
    
  } catch (error) {
    // The internet dropped! We caught the crash.
    // Instead of a white screen of death, we show an elegant popup:
    showToast("Network connection lost. Please try again.");
    
  } finally {
    // No matter if it succeeded or failed, we MUST turn off the spinner!
    setLoading(false);
  }
}
```

## INTERVIEW
**Q: What is the difference between `throw "Error"` and `throw new Error("Error")`?**
A: In JavaScript, you can technically `throw` anything (a string, a number, or an object). However, throwing a raw string (`throw "Bad Data"`) is universally considered a terrible practice. When you throw a string, you only throw text. When you `throw new Error()`, the V8 engine generates a massive object that includes a `Stack Trace`—a detailed blueprint showing the exact file name and line number where the crash originated, which is critical for debugging.

**Q: Can you have a `try` block without a `catch` block?**
A: Yes, but only if you provide a `finally` block! A `try { } finally { }` structure is valid. However, because there is no `catch`, any errors thrown inside the `try` will still bubble up and crash the application. It simply guarantees that the `finally` cleanup code runs right *before* the application crashes.

**Q: If an asynchronous `setTimeout` crashes inside a `try/catch` block, will the `catch` intercept it?**
A: **No!** This is a classic asynchronous trap. 
```javascript
try {
  setTimeout(() => { throw new Error("Boom!"); }, 1000);
} catch (e) {
  console.log("Caught!");
}
```
Because the `setTimeout` callback goes to the Web APIs and eventually the Task Queue, the main `try/catch` block finishes executing and dies instantaneously. One second later, the callback hits the Call Stack, explodes, and the application crashes. The `catch` block was long gone. You must put the `try/catch` *inside* the callback, or use Promises/Async-Await.

## FEYNMAN CHECK

### Explain JS Error Handling Like I'm 10 Years Old
> JavaScript has two error worlds: SYNCHRONOUS (throw/try/catch/finally) and ASYNCHRONOUS (unhandled promise rejections, onerror, onunhandledrejection). Sync errors: `throw new Error('msg')` creates an Error with a stack trace; `try { ... } catch (e) { ... } finally { ... }` catches it. Types: Error (base), TypeError, RangeError, ReferenceError, SyntaxError, URIError, EvalError — extend Error for custom ones. Async: if a Promise rejects and nobody catches it, it fires window.onunhandledRejection (browser) or process.on('unhandledRejection') (Node). The non-obvious rule: ALWAYS throw Error OBJECTS, not strings — only Error objects have a stack trace, which is invaluable for debugging in production.

### 5 Deep Conceptual Questions

**Q1: When should you catch an error vs let it propagate?**
> **A:** Catch when you can RECOVER — translate to a safe default, retry, show user-friendly feedback, or convert to a domain error type. Let propagate when you cannot meaningfully recover (let the outer boundary handle it). The pattern: narrow try/catch scopes around only the operation that might fail, not entire functions. A global error boundary (React's ErrorBoundary, Express error middleware, process.on('unhandledRejection')) handles unrecoverable errors gracefully.

**Q2: Mental model for error wrapping?**
> **A:** When catching and re-throwing, ALWAYS add context: `throw new Error('Failed to load user profile', { cause: originalError })`. ES2022's Error `cause` option chains errors — the original stack trace is preserved. This produces "stacktrace chains" useful in production: you see "Failed to load user profile → cause: HTTP 404" instead of just "HTTP 404" with no call-site context.

**Q3: Most dangerous misconception?**
> **A:** Try/catch catches async errors automatically:
> ```js
> // ❌ Promise rejection in setTimeout is uncaught
> async function fetchData() {
>   try {
>     setTimeout(async () => {
>       const r = await fetch('/bad');  // rejection here
>       if (!r.ok) throw new Error('bad');
>     }, 0);
>   } catch (e) {
>     console.log('caught:', e);  // never runs
>   }
> }
>
> // ✅ Return/await the promise
> async function fetchData() {
>   try {
>     const r = await fetch('/bad');
>     if (!r.ok) throw new Error('bad');
>   } catch (e) {
>     console.log('caught:', e);  // works
>   }
> }
> ```

**Q4: How do custom error classes improve error handling?**
> **A:** Custom errors let you DISCRIMINATE error types: `catch (e) { if (e instanceof NetworkError) retry(); else if (e instanceof AuthError) redirect('/login'); else throw e; }`. Each custom error can carry structured data (status code, request URL, validation messages) beyond a string message. The classic pattern: an AppError base class with error codes, all domain errors extend it, global handler checks instanceof chain.

**Q5: FAANG-grade definition?**
> **A:** "JavaScript error handling spans synchronous throw/catch/finally (with typed Error subclasses and the ES2022 `cause` chain), asynchronous promise rejection (caught by .catch/.then(undefined, handler)/try-await or escaping to unhandledRejection global handlers), and cross-boundary error context via Error wrapping — with custom Error subclasses enabling type-discriminated recovery, and global process/window handlers as the safety net for unrecoverable errors."

#### Step 2 — Error Hierarchy
```js
// errors.js

// Base: all domain errors extend this
export class AppError extends Error {
  constructor(message, { code, cause, data } = {}) {
    super(message, { cause });
    this.name = this.constructor.name;
    this.code = code ?? 'APP_ERROR';
    this.data = data ?? {};
    if (Error.captureStackTrace) Error.captureStackTrace(this, this.constructor);
  }
  toJSON() {
    return { error: this.name, code: this.code, message: this.message, data: this.data };
  }
}

export class NetworkError extends AppError {
  constructor(message, { status, url, cause } = {}) {
    super(message, { code: 'NETWORK_ERROR', cause, data: { status, url } });
    this.status = status;
    this.url = url;
  }
}

export class AuthError extends AppError {
  constructor(message, reason = 'UNAUTHORIZED') {
    super(message, { code: reason });
  }
  get isExpired() { return this.code === 'TOKEN_EXPIRED'; }
}

export class ValidationError extends AppError {
  constructor(message, fields = {}) {
    super(message, { code: 'VALIDATION_FAILED', data: { fields } });
    this.fields = fields;
  }
}

export class NotFoundError extends AppError {
  constructor(resource, id) {
    super(`${resource} with id ${id} not found`, { code: 'NOT_FOUND', data: { resource, id } });
  }
}
```

#### Step 3 — Global Error Handler
```js
// handler.js
import { NetworkError, AuthError, ValidationError, NotFoundError, AppError } from './errors.js';

export function handleError(error, context = {}) {
  // Type-discriminated recovery
  if (error instanceof AuthError) {
    if (error.isExpired) refreshToken();
    else redirectToLogin();
    return;
  }
  if (error instanceof NetworkError) {
    const { status, url } = error.data;
    logMetric({ event: 'network_error', status, url });
    showRetryUI();
    return;
  }
  if (error instanceof ValidationError) {
    showFieldErrors(error.fields);
    return;
  }
  if (error instanceof NotFoundError) {
    showEmptyState(error.data.resource);
    return;
  }
  if (error instanceof AppError) {
    logError(error, 'unhandled_app_error', context);
    showGenericError();
    return;
  }
  // Unknown / non-domain error — escalate
  logError(error, 'unhandled_unknown_error', context);
  throw error;
}

// Global async boundary
if (typeof window !== 'undefined') {
  window.onunhandledrejection = (e) => { handleError(e.reason); e.preventDefault(); };
}
process.on?.('unhandledRejection', (reason) => handleError(reason));
```

#### Step 4 — Error Wrapping + cause
```js
// Using Error.cause for context chains
async function loadUserProfile(userId) {
  let rawUser;
  try {
    const res = await fetch(`/api/users/${userId}`);
    if (res.status === 404) throw new NotFoundError('User', userId);
    if (res.status === 401) throw new AuthError('Session expired', 'TOKEN_EXPIRED');
    if (!res.ok) throw new NetworkError(`API error`, { status: res.status, url: res.url });
    rawUser = await res.json();
  } catch (e) {
    if (e instanceof AppError) throw e;   // already typed
    throw new NetworkError('Failed to load user profile', { cause: e });  // wrap + preserve
  }
  return rawUser;
}

// Caller sees chained context
try {
  await loadUserProfile(999);
} catch (e) {
  console.error(e.message);          // "User with id 999 not found"
  console.error(e.cause?.message);   // original error if wrapped
  console.log(e.toJSON());
}
```
